AnyConstraint

public struct AnyConstraint<T, E> : Constraint where E : Error

A type-erased Constraint.

enum Failure: Error {
    case notEven
}

let constraint = BlockConstraint<Int, Failure> {
    $0 % 2 == 0
} errorBuilder: {
    .notEven
}

let anyConstraint = AnyConstraint(constraint)
anyConstraint.evaluate(with: 3)
