AnyPredicate

public struct AnyPredicate<T> : Predicate

A type-erased Predicate.

let odd = BlockPredicate<Int> {
   $0 % 2 != 0
}

let anyOdd = AnyPredicate<Int>(odd)
let isOdd = anyOdd.evaluate(with: 3)
